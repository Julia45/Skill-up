Привязка классов

Использование ngClass

Набор классов в виде строки
[ngClass] = "'border border-danger bg-success text-warning rounded p-3 mt-5'"
Набор классов в виде массива
[ngClass] = "['border', 'border-info', 'bg-warning', 'text-light', 'rounded', 'p-3', 'mt-4']"
Набор классов в виде объекта
[ngClass] = "{'border': true, 'border-success': true, 'bg-info': true, 'text-light': true, 'rounded':true, 'p-3': true, 'mt-4': true{"
Набор классов в виде объекта
[ngClass] = "{'border border-warning bg-danger text-warning rounded p-3 mt-4': true{"
Набор классов в виде надстроечного объекта
[ngClass] = "settings"
Наборы классов по условию
[ngClass] = "isChecked ? checkedClasses : uncheckedClasses"
